아래 코드를 기존 report.html에 추가하면 로그인 보호와 역할별 UI가 적용됩니다.

<head> 태그 안에 추가 (Firebase SDK 로드)

<!-- Firebase Auth Guard -->
<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";
import { getAuth, onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js";
import { getDatabase, ref, get } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-database.js";

const firebaseConfig = {
  apiKey: "AIzaSyAtdO1c-bvADTGYdIEAfGN_Pc36g2hL3_4",
  authDomain: "mobilog-report.firebaseapp.com",
  databaseURL: "https://mobilog-report-default-rtdb.asia-southeast1.firebasedatabase.app",
  projectId: "mobilog-report",
  storageBucket: "mobilog-report.firebasestorage.app",
  messagingSenderId: "135384880017",
  appId: "1:135384880017:web:568e70398e403a0116c030"
};

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getDatabase(app);

// 페이지 로드 시 인증 확인
onAuthStateChanged(auth, async (user) => {
  if (!user) {
    // 비로그인 → 로그인 페이지로
    window.location.href = 'login.html';
    return;
  }
  
  // 역할 확인
  const snap = await get(ref(db, `users/${user.uid}`));
  const userData = snap.val() || {};
  const role = userData.role || 'user';
  const name = userData.name || user.email;

  // 현재 사용자 정보를 전역으로 저장
  window.currentUserRole = role;
  window.currentUserName = name;
  window.currentUserEmail = user.email;

  // UI 업데이트
  const nameEl = document.getElementById('userNameDisplay');
  if (nameEl) nameEl.textContent = name;

  const roleEl = document.getElementById('userRoleDisplay');
  if (roleEl) roleEl.textContent = role === 'admin' ? '관리자' : '일반 직원';

  // 관리자 전용 요소 표시
  document.querySelectorAll('.admin-only').forEach(el => {
    el.style.display = role === 'admin' ? '' : 'none';
  });

  // 일반 직원 전용 요소 표시
  document.querySelectorAll('.user-only').forEach(el => {
    el.style.display = role === 'user' ? '' : 'none';
  });
});

// 로그아웃 함수 (전역)
window.logout = async function() {
  await signOut(auth);
  window.location.href = 'login.html';
};
</script>

역할별 UI 제어 (선택사항)

특정 요소를 관리자만 보이게 하려면 class="admin-only"를 추가하세요:

<div class="admin-only" style="display:none;">전체 팀원 데이터</div>

<!-- 일반 직원만 볼 수 있는 요소 -->

첫 관리자 계정 만들기

admin.html을 사용하기 위해 최초 관리자 계정을 Firebase Console에서 직접 만들어야 합니다:

1. Firebase Console → Authentication → Users → 사용자 추가
2. 이메일 + 비밀번호로 계정 생성
3. 생성된 UID 복사
4. Realtime Database → 데이터 추가:

users/
  └── [복사한 UID]/
        ├── name: "관리자 이름"
        ├── email: "admin@example.com"
        └── role: "admin"    ← 반드시 "admin"으로
✅ 이후부터는 admin.html에서 팀원 계정을 직접 생성·관리할 수 있습니다!
